# 对表格增加值

function insert(tableName, data) {
    var key = [];
    var value = [];
    for (const i in data) {
        if (data.hasOwnProperty(i)) {
            key.push(i)
            value.push(JSON.stringify(data[i]))
        }
    }
    return `insert into ${tableName}(${key.toString()}) values(${value.toString()})`
}
