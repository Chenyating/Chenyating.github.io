# 公有私有方法

function Foo(){
    Foo.a=function(){
        console.log(1)
    }
    this.a=function(){
        console.log(2)
    }
}
// 吧Foo当做类，在原型上设置实例公有属性方法
Foo.prototype.a=function(){
    console.log(4)
}
//把foo当做普通对象设置私有属性方法 
Foo.a=function(){
    console.log(3)
}

Foo.a();//3
new obj=new Foo();//会执行Foo();替换之前的
obj.a();
Foo.a();

// 结果为：3，2，1

# 同步和异步

# 同步

  • 同步会阻塞后面的代码执行
  • 没有回调

# 异步

  • 异步是非阻塞
  • 异步一定有回调

# 单线程

  • 代码从上到下执行
  • 同步& 一步，同步任务会导致阻塞
  • 同步代表，alert()，console.log()，赋值语句
